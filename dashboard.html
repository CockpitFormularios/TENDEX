// CORRIGIR A FUNÇÃO renderCards - garantir que o ID seja usado corretamente function renderCards(filterText = "") { const container = document.getElementById('itemsContainer'); const lowerFilter = filterText.toLowerCase(); const filtered = currentItems.filter(item => item.name.toLowerCase().includes(lowerFilter) || item.code.toLowerCase().includes(lowerFilter) || (item.sub_localizacao && item.sub_localizacao.toLowerCase().includes(lowerFilter)) || (item.tamanho && item.tamanho.toLowerCase().includes(lowerFilter))); if (filtered.length === 0) { container.innerHTML = `
✨ Nenhum item encontrado.
`; return; } const dataAtual = new Date(); const mesAtualIndex = dataAtual.getMonth(); const mesAtualNome = monthNames[mesAtualIndex]; const mesAtualKey = monthKeys[mesAtualIndex]; container.innerHTML = filtered.map(item => { const arquivoMesAtual = item.files.find(f => f.category === mesAtualKey); let inspecaoMsg = ''; if (arquivoMesAtual) { if (arquivoMesAtual.inspecao_data) inspecaoMsg = `

✅ Inspeção ${mesAtualNome}: ${formatarData(arquivoMesAtual.inspecao_data)}

`; else inspecaoMsg = `

✅ Inspeção ${mesAtualNome}: realizada

`; } else inspecaoMsg = `

⚠️ Inspeção ${mesAtualNome}: pendente

`; const statusClass = item.status === 'active' ? 'status-active' : 'status-inactive'; const statusText = item.status === 'active' ? 'Ativo' : 'Inativo'; const pendingIcon = item.pending ? '⏳ Pendente' : ''; // Usar item.id diretamente (pode ser string ou número) return `

${escapeHtml(item.name)}${pendingIcon}

TAG: ${escapeHtml(item.code)}

${item.sub_localizacao ? `

📍 ${escapeHtml(item.sub_localizacao)}

` : ''}${item.tamanho ? `

📏 ${escapeHtml(item.tamanho)}

` : ''}
${statusText}
${inspecaoMsg}
`; }).join(''); document.querySelectorAll('.item-card').forEach(card => { const id = card.dataset.id; card.addEventListener('click', () => openDetailModal(id)); }); }
